DROP TABLE accounts;

CREATE TABLE players(
id BIGSERIAL NOT NULL PRIMARY KEY,
name VARCHAR(50) NOT NULL,
level INT NOT NULL,
exp INT NOT NULL,
job INT NOT NULL,
max_hp INT NOT NULL,
hp INT NOT NULL,
max_mp INT NOT NULL,
mp INT NOT NULL,
stats_alloc BYTEA,
skills_alloc BYTEA);
